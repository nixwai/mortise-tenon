控制元素排列顺序，能够同时包含子级元素的排列顺序。

基础用法


2
7
9
6
5
4
3
8
1
<script setup lang="ts">
import { MtSort } from '@mortise-tenon/components';
import { ref } from 'vue';

const list = ref([2, 7, 9, 6, 4, 3, 1]);

function handleClick() {
  list.value = list.value.reverse();
}
</script>

<template>
  <button @click="handleClick">
    点击
  </button>
  <br>
  <MtSort :sort-list="list">
    <div>
      <div v-for="i in 5" :key="i" :sort-key="i">
        {{ i }}
      </div>
    </div>
    <div :sort-key="6">
      6
    </div>
    <div>
      <div :sort-key="7">
        7
      </div>
      <div>
        8
      </div>
      <div :sort-key="9">
        9
      </div>
    </div>
  </MtSort>
</template>

<style scoped>
div {
  display: inline-block;
}
</style>

属性

参数说明类型默认值
sortList排列顺序(number|string)[][]
keyName自定义节点排序属性的key名称stringsort-key
